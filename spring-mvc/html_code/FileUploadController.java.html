9    import javax.servlet.http.HttpServletRequest;

12   import java.util.HashMap;

17   @Controller 
18   @RequestMapping(value = "/file") 
19   public class FileUploadController { 
20    
21       @RequestMapping(value = "/upload.do") 
22       @ResponseBody 
23       public Object handlerFileUpload(@RequestParam("filename") MultipartFile fileField, 
24                                       HttpServletRequest request) throws IOException { 
25           HashMap<String, Object> map = new HashMap<>(); 
26    
27           String path = request.getSession().getServletContext().getRealPath("/img"); 
28           String[] extensions = { "gif", "jpg", "png","jpeg" }; 
29           String filename = fileField.getOriginalFilename(); 
30           String extName = filename.substring(filename.lastIndexOf(".")+1).toLowerCase(); 
31    
32           System.out.println("path = " + path); 
33           System.out.println("filename = " + filename); 
34           System.out.println("extName = " + extName); 
35    
36           if (fileField.getSize() > 0) { 
37    
38               /** 
39                *  判断上传的图片格式是否符合要求 
40                */ 
41               boolean isLegal = false; 
42               for (int i = 0; i < extensions.length; i++) { 
43                   if (extName.equals(extensions[i])) { 
44                       isLegal = true; 
45                       break; 
46                   } 
47               } 
48    
49               if (isLegal) { 
50                   // 图片符合格式要求保存下来 
51                   File file = new File(path, filename); 
52                   fileField.transferTo(file); 
53    
54                   map.put("status", true); 
55                   map.put("msg", "上传成功"); 
56               } else { 
57                   map.put("status", false); 
58                   map.put("msg", "请不要上传gif、jpg、png、jpeg以外格式的图片"); 
59               } 
60    
61           } 
62           return map; 
63       }

65   }
