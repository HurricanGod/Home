69       function checkFileSize(id, maxSize){ 
70           var  browserCfg = {}; 
71           var ua = window.navigator.userAgent; 
72           if (ua.indexOf("MSIE")>=1){ 
73               browserCfg.ie = true; 
74           }else if(ua.indexOf("Firefox")>=1){ 
75               browserCfg.firefox = true; 
76           }else if(ua.indexOf("Chrome")>=1){ 
77               browserCfg.chrome = true; 
78           } 
79    
80           var obj_file = $(id); 
81           if(obj_file.value==""){ 
82               alert("请先选择上传文件"); 
83               return false; 
84           } 
85           var size = obj_file[0].files[0].size; 
86           if(size < maxSize){ 
87               return true; 
88           } else { 
89               return false; 
90           } 
91       }
